<script>
(function() {
  var K = "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";
  var chatHistory = [];
  var isLoading = false;
  var SYSTEM_PROMPT = "You are Ankit Study AI Assistant — a smart, friendly tutor for D.Pharmacy students in India. Subjects: Pharmaceutics, Pharmacology, Pharmacognosy, Human Anatomy & Physiology, Social Pharmacy, Pharmaceutical Chemistry. Answer in simple Hinglish or English. Give exam-focused bullet point answers. Bold key terms with **term**. Keep answers under 300 words.";

  function addMsg(text, isUser) {
    var box = document.getElementById('ankit-chat-box');
    var div = document.createElement('div');
    div.className = 'ankit-msg ' + (isUser ? 'user' : 'ai');
    var label = document.createElement('div');
    label.className = 'msg-label';
    label.textContent = isUser ? '👤 Tum' : '🤖 AI';
    div.appendChild(label);
    var content = document.createElement('div');
    content.innerHTML = text.replace(/\*\*(.*?)\*\*/g,'<strong>$1</strong>').replace(/\n/g,'<br>');
    div.appendChild(content);
    box.appendChild(div);
    box.scrollTop = box.scrollHeight;
  }

  function addTyping() {
    var box = document.getElementById('ankit-chat-box');
    var div = document.createElement('div');
    div.className = 'ankit-msg ai';
    div.id = 'ankit-typing';
    var label = document.createElement('div');
    label.className = 'msg-label';
    label.textContent = '🤖 AI';
    div.appendChild(label);
    var dots = document.createElement('div');
    dots.className = 'typing-dots';
    dots.innerHTML = '<span></span><span></span><span></span>';
    div.appendChild(dots);
    box.appendChild(div);
    box.scrollTop = box.scrollHeight;
  }

  function removeTyping() {
    var t = document.getElementById('ankit-typing');
    if (t) t.remove();
  }

  async function callAI(userMsg) {
    var messages = [{ role: "system", content: SYSTEM_PROMPT }];
    var recent = chatHistory.slice(-8);
    for (var i = 0; i < recent.length; i++) messages.push(recent[i]);
    messages.push({ role: "user", content: userMsg });

    var response = await fetch("https://openrouter.ai/api/v1/chat/completions", {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        "Authorization": "Bearer " + K,
        "HTTP-Referer": "https://blogger.com",
        "X-Title": "Ankit Study AI"
      },
      body: JSON.stringify({
        model: "meta-llama/llama-3.1-8b-instruct:free",
        messages: messages,
        max_tokens: 800,
        temperature: 0.7
      })
    });

    var data = await response.json();
    if (!response.ok) throw new Error(JSON.stringify(data.error || data));
    if (!data.choices || !data.choices[0]) throw new Error("Response nahi mila");
    return data.choices[0].message.content;
  }

  window.ankitSend = async function() {
    if (isLoading) return;
    var input = document.getElementById('ankit-input');
    var msg = input.value.trim();
    if (!msg) return;
    input.value = '';
    isLoading = true;
    document.getElementById('ankit-send-btn').disabled = true;
    addMsg(msg, true);
    addTyping();
    try {
      var reply = await callAI(msg);
      removeTyping();
      addMsg(reply, false);
      chatHistory.push({ role: "user", content: msg });
      chatHistory.push({ role: "assistant", content: reply });
      if (chatHistory.length > 16) chatHistory = chatHistory.slice(-16);
    } catch(e) {
      removeTyping();
      addMsg("❌ Error: " + e.message, false);
    }
    isLoading = false;
    document.getElementById('ankit-send-btn').disabled = false;
    input.focus();
  };

  window.ankitQuick = function(q) {
    document.getElementById('ankit-input').value = q;
    ankitSend();
  };

  document.getElementById('ankit-input').addEventListener('keypress', function(e) {
    if (e.key === 'Enter') ankitSend();
  });

  setTimeout(function() {
    addMsg("Hello! 👋 Main hoon **Ankit Study AI** — tumhara D.Pharmacy exam helper!\n\nKoi bhi sawaal pucho — main yahan hoon! 📚\n\n💡 Neeche quick buttons bhi try karo!", false);
  }, 400);
})();
</script>
